683 auto* button_watcher = new DigitalInputChange(

685 // Create a debounce transform

688 auto* debounce = new DebounceInt(kDebounceDelay, kConfigPathDebounceSwitch);

690 ->set_title("MagCal Button Debounce")
691 ->set_description("Debounce delay (ms) for Magnetic Calibration save button.")
692 ->set_sort_order(1000);
693 // Define the action taken when button is active and debounce has elapsed.
694 // Provide it with the context of orientation_sensor so it can access save fcn.
695 auto save_mcal_function = [orientation_sensor](int input) {
696 if (input == SWITCH_ACTIVE_STATE) {
697 orientation_sensor->sensor_interface_->SaveMagneticCalibration();
698 debugI("Magnetic Calibration values saved");
699 }
700 };
701 auto* button_consumer = new LambdaConsumer<int>(save_mcal_function);
702 // Connect the button -> debounce -> save magnetic calibration fcn
703 button_watcher->connect_to(debounce)->connect_to(button_consumer);
