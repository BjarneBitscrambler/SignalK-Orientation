222 void OrientationValues::Update() {
223  //check whether magnetic calibration has been requested to be saved
224  if( 1 == save_mag_cal_ ) {
225  orientation_sensor_->sensor_interface_->InjectCommand("SVMC");
226  save_mag_cal_ = 0; // set flag back to zero so we don't repeat save
227  }
228  //check which type of parameter is requested, and pass it on
229  switch (value_type_) {
230  case (kCompassHeading):
231  output = orientation_sensor_->sensor_interface_->GetHeadingRadians();
232  break;
233  case (kRoll):
234  output = orientation_sensor_->sensor_interface_->GetRollRadians();
235  break;
236  case (kPitch):
237  output = orientation_sensor_->sensor_interface_->GetPitchRadians();
238  break;
239  case (kAccelerationX):
240  output = orientation_sensor_->sensor_interface_->GetAccelXMPerSS();
241  break;
242  case (kAccelerationY):
243  output = orientation_sensor_->sensor_interface_->GetAccelYMPerSS();
244  break;
245  case (kAccelerationZ):
246  output = orientation_sensor_->sensor_interface_->GetAccelZMPerSS();
247  break;
248  case (kRateOfTurn):
249  output = orientation_sensor_->sensor_interface_->GetTurnRateRadPerS();
250  break;
251  case (kRateOfPitch):
252  output = orientation_sensor_->sensor_interface_->GetPitchRateRadPerS();
253  break;
254  case (kRateOfRoll):
255  output = orientation_sensor_->sensor_interface_->GetRollRateRadPerS();
256  break;
257  case (kTemperature):
258  output = orientation_sensor_->sensor_interface_->GetTemperatureK();
259  break;
260  default:
261  return; //skip the notify(), due to unrecognized value type
262  }
263  if (orientation_sensor_->sensor_interface_->IsDataValid()) {
264  notify(); // only pass on the data if it is valid
265  }
266 } // end Update()
